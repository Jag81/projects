        var autocomplete_req={cmd:"find",field:'.Data.First_Name',table:m.ListTb,options:{},skip:0,limit:10}
    var autocomplete_callback=function(items){}
    var autocomplete_list=function(records){
        var items=[];
        for(var i=0;i<records.length;i++){
            var obj={};
            obj.label=records[i].Data["First_Name"]+" "+records[i].Data["Last_Name"];
            items.push(obj);
        }
        return items;
    }
    var wait1=function(){
        $vm.autocomplete( $('input[name=Name'),autocomplete_req,autocomplete_list,autocomplete_callback);
    }
    var I=0; var loop_1=setInterval(function (){
        if($vm['jquery-ui-min-js']!=undefined){  clearInterval(loop_1); wait1(); }
        I++; if(I>50){ clearInterval(loop_1); alert("jquery-ui.min.js is not installed."); }
